119 template<size_t32 S>
120 void* endian::swap_bytes( void* swapMe )
121 {
122  u64 indexLow = 0;
123  u64 indexHigh = S-1;
124  u8* bytes = static_cast<u8*>( swapMe );
125 
126  while( indexLow < indexHigh )
127  {
128  u8 tmp = bytes[indexLow];
129  bytes[indexLow] = bytes[indexHigh];
130  bytes[indexHigh] = tmp;
131  ++indexLow;
132  --indexHigh;
133  }
134  return swapMe;
135 }
