194  {
195  SCOPE_MUTEX( &_mutex );
196  Allocations* temp_allocation = reinterpret_cast<Allocations*>( crap::memory::MemoryManager::instance().allocate( sizeof(Allocations) ) );
197  temp_allocation->object = reinterpret_cast<pointer>(crap::memory::MemoryManager::instance().allocate(count * sizeof(T)));
198 // std::cout << "object is " << temp_allocation->object << std::endl;
199  temp_allocation->next = 0;
200  temp_allocation->previous = 0;
201 
202  if( _size == 0 )
203  {
204  _start = temp_allocation;
205  _last = temp_allocation;
206  }
207  else
208  {
209  _last->next = temp_allocation;
210  temp_allocation->previous = _last;
211  _last = temp_allocation;
212  }
213  _size += count;
214  return temp_allocation->object;
215  }
