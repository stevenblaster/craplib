151 // allocate memory
152 void* memory_pool::allocate(size_t32 size)
153 {
154  //size to allocate is space + header size
155  size_t32 allocation_size = sizeof(memory_block) + size;
156 
157  //assert there is space at all AND if there's a block being big enough
158  CRAP_ASSERT_DEBUG( _current_size >= allocation_size, "Not enough space for allocation" );
159  CRAP_ASSERT_DEBUG( _biggest_block->free_memory >= allocation_size, "Not enough space for allocation" );
160 
161  //pointer to the block containing enough free space, init. with zero
162  memory_block* found_block = 0;
163 
164  //check if the very first block has enough space - if it doesnt...
165  if( _block_start->free_memory < allocation_size )
166  {
167  //the parsed memory - init. with the complete size of the first block
168  size_t32 parsed_memory = sizeof(memory_block) + _block_start->used_memory + _block_start->free_memory;
169 
170  //this size will be updated each step of folling loop - should extend pool's size
171  while( parsed_memory < _size )
172  {
173  //pointer to position we expect the next block
174  memory_block* parsed_block = (memory_block*)(reinterpret_cast<u8*>(_block_start) + parsed_memory);
175 
176  //check if this block has enough free space for the wanted allocation
177  if( parsed_block->free_memory >= allocation_size )
178  {
179  //it has enough space, lets set the found_block variable and break the loop
180  found_block = parsed_block;
181  break;
182  }
183  else
184  {
185  //not enough space, lets update parsed memory for the next step in loop
186  parsed_memory += sizeof(memory_block) + parsed_block->used_memory + parsed_block->free_memory;
187  }
188  }
189  }
190  else
191  {
192  //the very first block has enough space
193  found_block = _block_start;
194  }
195 
196  //assert a block was found
197  CRAP_ASSERT_DEBUG( found_block != 0, "No free block with space enough found" );
198 
199  //lets set up the new block on free space's position
200  memory_block* new_block = (memory_block*)(reinterpret_cast<u8*>(found_block + 1) + found_block->used_memory);
201 
202  //insert offset to previous block
203  new_block->offset_prev = sizeof(memory_block) + found_block->used_memory;
204 
205  //insert used (not allocated, since header is not used) space of block
206  new_block->used_memory = size;
207 
208  //reduce free memory bye allocated size (used size + header size)
209  new_block->free_memory = found_block->free_memory - allocation_size;
210 
211  //previous block has no free memory left (new one overtook rest)
212  found_block->free_memory = 0;
213 
214  //if used block was the biggest, lets update that one
215  if( found_block == _biggest_block )
216  {
217  update_biggest_block();
218  }
219 
220  //return pointer to used space
221  return new_block + 1;
222 }

224 // update biggest block (internal)
225 void memory_pool::update_biggest_block(void)
226 {
227  //init. parsed block with starting block
228  memory_block* parsed_block = _block_start;
229 
230  //init. parsed memory with size of first block
231  size_t32 parsed_memory = sizeof(memory_block) + parsed_block->used_memory + parsed_block->free_memory;
232 
233  //parse whole block
234  while( parsed_memory < _size )
235  {
236  //check next block...
237  memory_block* check_block = (memory_block*)(reinterpret_cast<u8*>(_block_start) + parsed_memory);
238 
239  //if there's more free space, lets update the parsed pointer
240  if( parsed_block->free_memory < check_block->free_memory )
241  parsed_block = check_block;
242 
243  //else lets update parsed memory
244  parsed_memory += sizeof(memory_block) + check_block->used_memory + check_block->free_memory;
245  }
246 
247  //update biggest block pointer
248  _biggest_block = parsed_block;
249 }

283 // deallocate
284 void memory_pool::deallocate(void* address)
285 {
286  //assert the pointer is in range
287  CRAP_ASSERT_DEBUG( is_in_range(address) == true, "Address for deallocation out of range" );
288 
289  //get block info
290  memory_block* delete_block = (memory_block*) address;
291 
292  //parse previous block
293  memory_block* previous_block = (memory_block*)(reinterpret_cast<u8*>(delete_block) - delete_block->offset_prev);
294 
295  //check if previous block is still in range
296  CRAP_ASSERT_DEBUG( previous_block >= _block_start, "Previous offset of deallocation pointer out of range" );
297 
298  //crosscheck if data of previous block is correct
299  bool check_offset = reinterpret_cast<memory_block*>(reinterpret_cast<u8*>(previous_block + 1) +
300  previous_block->used_memory + previous_block->free_memory) == delete_block;
301 
302  // if crosscheck failed - abort
303  CRAP_ASSERT_DEBUG( check_offset == true , "Offset check error, previous offset doesn't match" );
304 
305  //update free memory of previous block with complete size of deleted block
306  previous_block->free_memory += sizeof(memory_block) + delete_block->used_memory + delete_block->free_memory;
307 
308  //update biggest free block
309  update_biggest_block();
310 }
