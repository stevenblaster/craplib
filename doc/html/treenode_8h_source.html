276  // node on the right
277  else if( delete_node->sub_node[left] == 0 && delete_node->sub_node[right] != 0 )

280  {

288  }
289  else
290  return 0;
291 
292  return delete_node;
293  }
294  // node on the left
295  else if( delete_node->sub_node[left] != 0 && delete_node->sub_node[right] == 0 )

543 {
544  if( low_node == 0 || high_node == 0 )
545  return 0;
