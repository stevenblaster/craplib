278 template <class T, class C, class A>
279 bool binary_tree<T,C,A>::remove( const_reference key )
280 {
281  node_pointer searched_node = _root->search_recursive(_root, key);
282  if( searched_node != 0 )
283  {
284  if(searched_node == _root)
285  {
286  node_pointer root_left = _root->sub_node[node_type::left];
287  node_pointer root_right = _root->sub_node[node_type::right];
288 
289  node_pointer root_predecessor = _root->predecessor_iterative(_root);
290  node_pointer root_successor = _root->successor_iterative(_root);
291 
292  searched_node = _root->unlink(searched_node);
293  if( searched_node == 0 )
294  return false;
295 
296  if(root_left->sub_node[node_type::right] == root_right)
297  {
298  _root = root_left;
299  }
300  else if(root_right->sub_node[node_type::left] == root_left)
301  {
302  _root = root_left;
303  }
304  else if(_root == searched_node)
305  return false;
306 
307  }
308  else
309  {
310  searched_node = _root->unlink(searched_node);
311  if( searched_node == 0 )
312  return false;
313  }
314  _allocator.deallocate(searched_node);
315  --_size;
316  return true;
317  }
318  return false;
319 }
