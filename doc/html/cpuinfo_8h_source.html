101  static inline i64 xgetbv (i32 ctr)
102  {
103  #if defined (CRAP_COMPILER_INTEL) // defined (CRAP_COMPILER_VC) ||
104  return _xgetbv(ctr);
105  #elif defined(CRAP_COMPILER_GCC) && !defined(CRAP_PLATFORM_MAC)
106  u32 a, d;
107  __asm("xgetbv" : "=a"(a),"=d"(d) : "c"(ctr) : );
108  return a | (u64(d) << 32);
109  #else
110  u32 a, d;
111  __asm {
112  mov ecx, ctr
113  _emit 0x0f
114  _emit 0x01
115  _emit 0xd0 ; // xgetbv
116  mov a, eax
117  mov d, edx
118  }
119  return a | (u64(d) << 32);
120  #endif
121  }

124  void simd_detect(void)
125  {
126  _detected = None;
127  i32 abcd[4] = {0,0,0,0};
128  CRAP_CPUID(0, abcd);
129 
130  // no further cpuid function supported
131  if (abcd[0] == 0)
132  return;
133 
134  // call cpuid function 1 for feature flags
135  CRAP_CPUID(1,abcd);
136  if ((abcd[3] & (1 << 0)) == 0 || (abcd[3] & (1 << 23)) == 0 || (abcd[3] & (1 << 15)) == 0 || (abcd[3] & (1 << 24)) == 0 || (abcd[3] & (1 << 25)) == 0)
137  return; // no floating point, no MMX, no conditional move, no FXSAVE, no SSE
138 
139  // 1: SSE supported
140  _detected = SSE;
141  if ((abcd[3] & (1 << 26)) == 0)
142  return; // no SSE2
143 
144  // 2: SSE2 supported
145  _detected = SSE2;
146  if ((abcd[2] & (1 << 0)) == 0)
147  return; // no SSE3
148 
149  // 3: SSE3 supported
150  _detected = SSE3;
151  if ((abcd[2] & (1 << 9)) == 0)
152  return; // no SSSE3
153 
154  // 4: SSSE3 supported
155  _detected = SSSE3;
156  if ((abcd[2] & (1 << 19)) == 0)
157  return; // no SSE4.1
158 
159  // 5: SSE4.1 supported
160  _detected = SSE41;
161  if ((abcd[2] & (1 << 23)) == 0 || (abcd[2] & (1 << 20)) == 0 )
162  return; // no POPCNT, no SSE4.2
163 
164  // 6: SSE4.2 supported
165  _detected = SSE42;
166  if ((abcd[2] & (1 << 27)) == 0 || (xgetbv(0) & 6) != 6 || (abcd[2] & (1 << 28)) == 0)
167  return; // no OSXSAVE, AVX not enabled in O.S., no AVX
168 
169  // 7: AVX supported
170  _detected = AVX;
171 
172  // call cpuid leaf 7 for feature flags
173  CRAP_CPUID(7, abcd);
174 
175  if ((abcd[1] & (1 << 5)) == 0)
176  return; // no AVX2
177 
178  // 8: AVX2 supported
179  _detected = AVX2;
180  return;
181  }
